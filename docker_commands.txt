Docker Networking 

docker network ls                            -- > list network

docker network inspect my-nginx-networking   -- > inspect network 

docker network create my-nginx-networking    -- > create network 

docker network rm my-nginx-networking        -- > remove network

docker network prune                         -- > remove unused network

docker run -d --name nginx --network \

	my-nginx-networking my-nginx            -- > run container in network 
	
docker network connect my-nginx-networking \
 
	nginx                                   -- > connect existing container
	
docker network disconnect my-nginx-networking \
	
	nginx                                   -- > disconnect container
	
docker network create --subnet 192.168.1.0/24 \ 

	my-nginx-networking                      -- > create network with subnet
	
docker network create -d bridge  \

	my-nginx-networking                      -- > create network with drive
	
docker run --network host nginx              -- > use host network 

docker run --network none nginx              -- > disable networking

docker inspect my-nginx | grep IPAddress     --> check container ip by using debugging

docker volume 


docker create volume nginx-vol              -- > create volume

docker volume ls                            -- > list volume

docker volume inspect nginx-vol             -- > inspect volume

docker volume rm nginx-vol                  -- > remove volume

docker volume prune                         -- > remove unused volume

docker run -d \
--name nginx \
-v nginx-vol:/data \
busybox sleep 3600                          -- > run container with volume 

docker run -d \
--name nginx \
--mount source=nginx-vol,target=/data \
busybox sleep 3600                         -- > run container with volume (--mount recommended)


docker volume create --driver local \

	nginx-vol                              -- > create volume with drive 
	
	
docker run -d \
--mount source=nginx-vol, \ 
target=/data,readonly \
my nginx                                   -- > use read only volume


docker inspect nginx_containers01          -- > check volume usage in container 

docker volume inspect my-nginx             -- > check volume path on host

Container Security

docker run -u 1001 nginx                 -- > run containers as non-root user 

docker run -- read-only nginx            -- > read only filesystem 

docker run --cap-drop ALL nginx          -- > drop all linux capabilities

docker run --cap-add NET_ADMIN nginx     -- > add only required capability

docker run -- memory=512 --cpus=1 nginx  -- > limit memory & cpu 

docker run --security-opt no-new-privileges \

	nginx                                 -- > Prevent privilege escalation

docker compose up:    Builds (if necessary), creates, and starts all defined services at once.
docker compose up -d: Runs the entire stack in the background (detached mode).
docker compose down:  Stops and removes all containers and networks created by the Compose file.
docker compose ps:    Lists the status of the services currently running in your stack.
docker compose logs:  Displays the aggregated log output from all services for easier debugging
